import java.util.Scanner;

public class NumberGuessingGame {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        boolean playAgain = true;
        int totalScore = 0;

        System.out.println("=== NUMBER GUESSING GAME ===");

        while (playAgain) {

            int secretNumber = 56;   
            int maxAttempts = 7;
            int attempts = 0;
            boolean guessed = false;

            System.out.println("\nGuess the secret number!");
            System.out.println("You have " + maxAttempts + " chances.");

            while (attempts < maxAttempts) {

                System.out.print("Enter your guess: ");
                int guess = sc.nextInt();
                attempts++;

                int diff = Math.abs(guess - secretNumber);

                if (guess == secretNumber) {
                    System.out.println(" Correct! You found the number.");
                    guessed = true;
                    totalScore += 10;
                    break;
                }

                
                else if (diff <= 2) {
                    System.out.println(" Very close!");
                }

                
                else if (diff <= 7) {
                    System.out.println("Close!");
                }

               
                else if (diff <= 16) {
                    if (guess < secretNumber) {
                        System.out.println(" Low!");
                    } else {
                        System.out.println(" High!");
                    }
                }

            
                else {
                    if (guess < secretNumber) {
                        System.out.println(" Too Low!");
                    } else {
                        System.out.println(" Too High!");
                    }
                }
            }

            if (!guessed) {
                System.out.println("\n you used all attempts!");
                System.out.println("Correct number was: " + secretNumber);
            }

            System.out.println("Attempts used: " + attempts);
            System.out.println("Current Score: " + totalScore);

            System.out.print("\nDo you want to play again? (yes/no): ");
            String choice = sc.next().toLowerCase();

            if (!choice.equals("yes")) {
                playAgain = false;
            }
        }

        System.out.println("\nFinal Score: " + totalScore);
        System.out.println("Thank you for playing");

        sc.close();
    }
}
